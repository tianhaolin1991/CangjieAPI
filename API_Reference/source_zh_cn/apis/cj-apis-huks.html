ohos.security.huks（通用密钥库系统）

向应用提供密钥库能力，包括密钥管理及密钥的密码学操作等功能。 HUKS所管理的密钥可以由应用导入或者由应用调用HUKS接口生成。

导入模块

import ohos.security_huks.*

type Bytes

type Bytes = Array<UInt8>

为提高代码可读性，本模块使用 Bytes 作为 Array<UInt8> 的别名，下方文档同理。

enum HuksParamValue

public enum HuksParamValue {

示例：

import ohos.security_huks.*
func clean(keyAlias: String) {
    try {
        if (isKeyItemExist(keyAlias, HuksOptions.NONE)) {
            deleteKeyItem(keyAlias, HuksOptions.NONE)
        }
    } catch (e: BusinessException) {
        e.printStackTrace()
    }
}

func test_generate_key() {
    let keyAlias = "test_generate_key"
