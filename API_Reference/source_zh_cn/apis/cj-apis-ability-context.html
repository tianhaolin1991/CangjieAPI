API 参考

AbilityContext

AbilityContext是需要保存状态的Ability所对应的context。

AbilityContext模块提供允许访问特定Ability的资源的能力，包括对Ability的启动、停止的设置、获取caller通信接口、拉起弹窗请求用户授权等。

暂不支持混合开发模式。

导入模块

import ohos.ability.*

使用说明

在使用AbilityContext的功能前，需要通过Ability子类实例获取。

import ohos.ability.*

var globalAbilityContext: Option<AbilityContext> = Option<AbilityContext>.None

class MainAbility <: Ability {
    public init() {
        super()
        registerSelf()
    }

    public override func onCreate(want: Want, launchParam: LaunchParam): Unit {
        globalAbilityContext = Option<AbilityContext>.Some(this.context)
    }
}

let ABILITY_REGISTER_RESULT = Ability.registerCreator("MainAbility", createAbility)

type StageContext

public type StageContext

保存状态的Ability所对应的context。支持混合开发和纯仓颉开发模式。

示例：

纯仓颉开发模式获取。

// main_ability.cj
import ohos.ability.*
var globalAbilityContext: Option<AbilityContext> = Option<AbilityContext>.None

class MainAbility <: Ability {
    public init() {
        super()
        registerSelf()
    }

    public override func onCreate(want: Want, launchParam: LaunchParam): Unit {
        AppLog.info("MainAbility OnCreated.${want.abilityName}")
        globalAbilityContext = Option<AbilityContext>.Some(this.context)
        match (launchParam.launchReason) {
            case LaunchReason.START_ABILITY => AppLog.info("START_ABILITY")
            case _ => ()
        }
    }

    public override func onWindowStageCreate(windowStage: WindowStage): Unit {
        AppLog.info("MainAbility onWindowStageCreate.")
        windowStage.loadContent("EntryView")
    }
}

// index.cj
import ohos.ability.*
func getContext(): AbilityContext {
    match(globalAbilityContext) {
        case Some(context) => context
        case _ => throw Exception("can not get globalAbilityContext.")
    }
}

let stageContext = getStageContext(getContext())

混合开发模式获取依赖于仓颉与ArkTS互操作库JSModule注册仓颉方法供ArkTS侧应用调用，函数注册方式参考仓颉与ArkTS互操作库。ArkTs侧传入的context为UIAbilityContext。

import ohos.ability.*
import ohos.ark_interop_helper.*

class Main {
    static init() {
        JSModule.registerFunc("setContext", { context => context.function(setContext) })
    }
}

static func setContext(context: JSContext, callInfo: JSCallInfo): JSValue {
    let jsEnv = context.getNapiEnv()
    let jsValue = arkTsValuetoNapiValue(jsEnv, callInfo[0])
    if (isStageMode(jsEnv, jsValue)) {
        // 建议用户缓存使用
        let stageContext: StageContext = getContextStageMode(jsEnv, jsValue)
    } else {
        AppLog.error("setContext: fail, not stage mode")
    }
    return context.undefined().toJSValue()
}

// MainAbility.ets
export default class MainAbility extends Ability {
    onCreate(want: Want, launchParam: AbilityConstant.LaunchParam) {
        // 参考互操作章节, 加载自定义仓颉互操作模块 cjLib 后
        // const cjLib: CustomLib = requireCJLib("libohos_app_cangjie_entry.so") as CustomLib
        // 调用 setContext 接口
        cjLib.setContext(this.context)
        ...
    }
}
